```html

	<nav>定义导航栏</nav>
	<footer>页面底部</footer>
	<article>定义文章</article>
	<section>定义区域</section>
	<input type="text" value="输入明星" list="star"/>

		<option >刘德华</option>
		<option >刘若英</option>
```
